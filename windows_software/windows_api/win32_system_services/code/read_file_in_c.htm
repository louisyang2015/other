#include <stdio.h>
#include <windows.h> //allows the constant "MAX_PATH"
void main()
{
    char filename[MAX_PATH]; //on my machine, MAX_PATH = 260
    char c;
    FILE *f;

    // get file name
    printf("Enter filename: ");
    gets(filename);

    // open the file
    f = fopen(filename, "r");
    if (f)
    {
        // read until eof
        while ((c = fgetc(f)) != EOF)
            printf("%c",c);
    }

    // close the file
    fclose(f);
}
