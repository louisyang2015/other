Input Signals Class:

module XXX_tb;
	class MyInSignal;
		int sig1, sig2;
		task set_to_x;
			sig1 = -1; sig2 = -1;
		endtask
		task set_to_0;
			sig1 = 0; sig2 = 0;
		endtask
		task set_to_1;
			sig1 = 1; sig2 = 1;
		endtask
	endclass
	
	MyInSignal is = new;
	
	task apply_input_signals;
		sig1 = int_to_bit(is.sig1);
		sig2 = int_to_bit(is.sig2);
	endtask
	
	function bit int_to_bit(int n);
		if(n == 0) return 0;
		else if (n == 1) return 1;
		else return $urandom_range(1, 0);
	endfunction
endmodule

module XXX_tb;
	class MyOutSignals;
		bit sig1, sig2;
	endclass
	
	MyOutSignals os = new;

	task run_and_check(int n);
		int i;
		for(i=0; i<n; i++) begin
			clk_1;
			assert(os.sig1 == sig1)
			else $display("something is wrong with: sig1 at time %t", $time);
			assert(os.sig2 == sig2)
			else $display("something is wrong with: sig2 at time %t", $time);
		end
	endtask
endmodule
