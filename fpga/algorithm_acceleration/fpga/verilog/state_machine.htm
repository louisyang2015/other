State Machine

Last updated April 4, 2011

// Declare state register
reg         [1:0]state;

// Declare states
parameter S0 = 0, S1 = 1, S2 = 2, S3 = 3;

// Determine the next state synchronously, based on the
// current state and the input
always @ (posedge clk or posedge reset) begin
    if (reset)
        state <= S0;
    else
        case (state)
            S0:
                if (in)
                begin
                    state <= S1;
                end
                else
                begin
                    state <= S1;
                end
            S1:
                if (in)
                begin
                    state <= S2;
                end
                else
                begin
                    state <= S1;
                end
            S2:
                if (in)
                begin
                    state <= S3;
                end
                else
                begin
                    state <= S1;
                end
            S3:
                if (in)
                begin
                    state <= S2;
                end
                else
                begin
                    state <= S3;
                end
        endcase
end
