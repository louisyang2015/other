CODE_SEG SEGMENT 'CODE'

LAB7 PROC FAR
    ASSUME CS:CODE_SEG, SS:STACK_SEG, DS:DATA_SEG

    ; to return to DEBUG program, put return address on the stack
    ; without these two lines, the program will not return
    ; to the right location upon exit
    PUSH DS
    MOV AX, 0
    PUSH AX

    ;establish data segment
    mov ax, DATA_SEG
    mov ds, ax

    ; PROGRAM CODE HERE

    RET
LAB7 ENDP

CODE_SEG ENDS

END LAB7
