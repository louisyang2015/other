BIOS Interrupt 10h, Service 2h

This service positions the hardware cursor on screen

For a normal screen, the upper left corner is (0, 0) and the lower right corner is (79, 24). Another possible height is to have 50 lines per screen.

mov AH, 2h

mov BH, 0h

mov DL, x_coordinate            ; 0 to 79

mov DH, y_coordinate            ; 0 to 24

int 10h

 

Service 10h in procedure form:

; expects x coordinate in DL, y coordinate in DH
; this procedure will set the cursor to (x, y)
GotoXY:
    mov AH, 02
    mov BH, 0
    int 10h
    ret

mov DX, [TextPos]    ; TextPos contains x, y coordinate values
call GotoXY
