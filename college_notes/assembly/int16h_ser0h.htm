BIOS Interrupt 16h, Service 0

This service reads a key

MOV AH, 0

INT 16H

;waits for a key to be pressed

;when a key is pressed, its corresponding ASCII code is placed into AL

In procedure form:

; wait for the user to press a key and
; return with a character code in AL
GetKey proc near:
    MOV AH, 0
    INT 16H
    RET
GetKey endp

call GetKey
cmp AL, 'q'    ;check to see if the "q" key is pressed
