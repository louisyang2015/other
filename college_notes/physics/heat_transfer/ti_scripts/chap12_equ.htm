tb12_1a()
Prgm

setMode("Angle","RADIAN")

Local x
Local y
Local l

Input "Enter X ", x
Input "Enter Y ", y
Input "Enter L ", l

Local xbar
Local ybar

x/l»xbar
y/l»ybar

Disp "xbar = "
disp xbar
Disp "ybar = "
disp ybar

Local a
(ln((1+xbar^2)*(1+ybar^2)/(1+xbar^2+ybar^2)))^1/2»a

Local b
xbar*(1+ybar^2)^(1/2)*tan°(xbar/(1+ybar^2)^(1/2))»b

Local c
ybar*(1+xbar^2)^(1/2)*tan°(ybar/(1+xbar^2)^(1/2))»c

Local d
ª1*xbar*tan°(xbar)-ybar*tan°(ybar)»d

Local f
2/(3.1415926536*xbar*ybar)*(a+b+c+d)»f

Disp "F = "
Disp f

EndPrgm

local rj

input "Enter L ", L

local Ri2
local Rj2

ri / L » Ri2
rj / L » Rj2

disp "Ri2 = "
disp Ri2

local S
1 + (1 + Rj2^2)/Ri2^2 » S

(1/2)*(S - (S^2 - 4 *(rj / ri)^2)^(1/2)) » f
disp "F = "

EndPrgm
tb12_1c()
Prgm

setMode("Angle","RADIAN")

local x
local y
local z

input "Enter X ", x
input "Enter Y ", y
input "Enter Z ", z

local H
local W

z/x » H
y/x » W

disp "H = "
disp H
disp "W = "
disp W

local a
W * tan°(1/W) + H * tan°(1/H) » a

local b
ª1*(H^2 + W^2)^(1/2) * tan°(1/(H^2 + W^2)^(1/2)) » b

local c
((1 + W^2)*(1 + H^2))/(1 + W^2 + H^2) » c

local d
((W^2 * (1 + W^2 + H^2))/((1 + W^2) * (W^2 + H^2)))^(W^2) » d

local e
((H^2 * (1 + H^2 + W^2))/((1 + H^2)*(H^2 + W^2)))^(H^2) » e

local F
(1/(3.1415926536 * W)) * (a + b + (1/4)* ln(c *d * e)) » F
disp "F = "
disp F

EndPrgm
